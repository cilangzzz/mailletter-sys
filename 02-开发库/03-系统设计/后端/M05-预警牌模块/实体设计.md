```java
 *
 * @author Claude
 */

@KeySequence("fz_warning_board_seq")

@EqualsAndHashCode(callSuper = true)

@NoArgsConstructor
@AllArgsConstructor

    /**
     * 主键ID
     */
    @TableId
    private Long id;

    /**
     * 预警牌名称
     */
    private String name;

    /**
     * 预警牌编码（唯一，创建后不可修改）

    private String code;

    /**
     * 站点类型（如：平安厅、信访室）
     */
    private String siteType;

    /**
     * 普通工作日时限（单位：工作日）
     */

    /**
     * 紧急时限（单位：工作日）

    private Integer urgentDays;

    /**
     * 特急时限（单位：工作日）
     */

    /**
     * 超紧急时限（单位：工作日）
     */
    private Integer superUrgentDays;

     * 预警阈值比例（百分比，如80表示时限的80%时预警）
     */

     */

    /**
     * 状态（0启用 1停用）

     * 枚举 {@link cn.iocoder.yudao.framework.common.enums.CommonStatusEnum}

     * 备注
     */

}
```

```java

@Getter
@AllArgsConstructor
public enum WarningLevelEnum implements ArrayValuable<String> {

    WARNING("warning", "预警", "即将超期"),
    OVERDUE("overdue", "超期", "超期1-3工作日"),
    SERIOUS_OVERDUE("serious_overdue", "严重超期", "超期4-7工作日"),
    CRITICAL_OVERDUE("critical_overdue", "极严重超期", "超期8-15工作日"),
    MAJOR_OVERDUE("major_overdue", "重大超期", "超期超过15工作日");

    public static final String[] ARRAYS = Arrays.stream(values()).map(WarningLevelEnum::getLevel).toArray(String[]::new);

    /**
     * 预警级别编码
     */
    private final String level;

    /**
     * 预警级别名称
     */
    private final String name;

    /**
     * 级别描述
     */
    private final String description;

    @Override
    public String[] array() {
        return ARRAYS;
    }

    /**
     * 根据超期天数确定预警级别
     *
     * @param overdueDays 超期工作日天数
     * @return 预警级别枚举
     */
    public static WarningLevelEnum getLevelByOverdueDays(Integer overdueDays) {
        if (overdueDays == null || overdueDays <= 0) {
            return WARNING; // 未超期时为预警级别
        }
        if (overdueDays <= 3) {
            return OVERDUE;
        }
        if (overdueDays <= 7) {
            return SERIOUS_OVERDUE;
        }
        if (overdueDays <= 15) {
            return CRITICAL_OVERDUE;
        }
        return MAJOR_OVERDUE;
    }

    /**

     */
    public static WarningLevelEnum valueOfLevel(String level) {
        return Arrays.stream(values())
                .filter(e -> e.getLevel().equals(level))
                .findFirst()
                .orElse(null);
```
