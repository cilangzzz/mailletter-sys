## 1. 实体概述

### 1.1 实体分层架构

本模块实体采用标准的三层架构设计：

| 层次 | 实体类型 | 命名规范 | 说明 |
|------|----------|----------|------|
| DO层 | 数据对象 | XxxDO | 与数据库表映射，继承TenantBaseDO |
| VO层 | 视图对象 | XxxReqVO/XxxRespVO | 前端交互数据传输 |
| Service层 | 业务对象 | XxxService | 业务逻辑处理 |

package cn.iocoder.yudao.module.fz.dal.mysql.statistics;

import cn.iocoder.yudao.framework.common.pojo.PageResult;
import cn.iocoder.yudao.framework.mybatis.core.mapper.BaseMapperX;
import cn.iocoder.yudao.framework.mybatis.core.query.LambdaQueryWrapperX;
import cn.iocoder.yudao.module.fz.controller.admin.statistics.vo.efficiency.EfficiencyStatisticsPageReqVO;
import cn.iocoder.yudao.module.fz.dal.dataobject.statistics.EfficiencyStatisticsDO;
import org.apache.ibatis.annotations.Mapper;

import java.util.List;

/**
 * 办理效率统计 Mapper
 */
@Mapper
public interface EfficiencyStatisticsMapper extends BaseMapperX<EfficiencyStatisticsDO> {

    default PageResult<EfficiencyStatisticsDO> selectPage(EfficiencyStatisticsPageReqVO reqVO) {
        return selectPage(reqVO, new LambdaQueryWrapperX<EfficiencyStatisticsDO>()
                .eqIfPresent(EfficiencyStatisticsDO::getUnitId, reqVO.getUnitId())
                .likeIfPresent(EfficiencyStatisticsDO::getUnitName, reqVO.getUnitName())
                .eqIfPresent(EfficiencyStatisticsDO::getParentUnitId, reqVO.getParentUnitId())
                .eqIfPresent(EfficiencyStatisticsDO::getStatPeriod, reqVO.getStatPeriod())
                .eqIfPresent(EfficiencyStatisticsDO::getStatYear, reqVO.getStatYear())
                .eqIfPresent(EfficiencyStatisticsDO::getStatMonth, reqVO.getStatMonth())
                .betweenIfPresent(EfficiencyStatisticsDO::getStatDate, reqVO.getStartDate(), reqVO.getEndDate())
                .orderByDesc(EfficiencyStatisticsDO::getCreateTime));
    }

    default List<EfficiencyStatisticsDO> selectList(EfficiencyStatisticsPageReqVO reqVO) {
        return selectList(new LambdaQueryWrapperX<EfficiencyStatisticsDO>()
                .eqIfPresent(EfficiencyStatisticsDO::getUnitId, reqVO.getUnitId())
                .likeIfPresent(EfficiencyStatisticsDO::getUnitName, reqVO.getUnitName())
                .eqIfPresent(EfficiencyStatisticsDO::getParentUnitId, reqVO.getParentUnitId())
                .eqIfPresent(EfficiencyStatisticsDO::getStatPeriod, reqVO.getStatPeriod())
                .eqIfPresent(EfficiencyStatisticsDO::getStatYear, reqVO.getStatYear())
                .eqIfPresent(EfficiencyStatisticsDO::getStatMonth, reqVO.getStatMonth())
                .betweenIfPresent(EfficiencyStatisticsDO::getStatDate, reqVO.getStartDate(), reqVO.getEndDate()));
    }

}
